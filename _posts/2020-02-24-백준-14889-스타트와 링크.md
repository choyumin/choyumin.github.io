## 14889.  스타트와 링크

---

### Code

```java
import java.io.*;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;

public class 스타트링크 {
	static int[] arr;
	static int[] sel;
	static StringBuilder sb;
	static int cnt = 0;
	static Queue<Integer> queue, sumqueue;
	static int sum = 0;
	static int[] arr2;
	static boolean[] visited;
	static int[] res;
	static int[][] map;
	static int tmp = 0;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

		String s = br.readLine();

		int N = Integer.parseInt(s);
		int M = N / 2;
		map = new int[N + 1][N + 1];

		for (int i = 1; i <= N; i++) {
			String[] str = br.readLine().split(" ");
			for (int j = 1; j <= N; j++) {
				map[i][j] = Integer.parseInt(str[j - 1]);
			}
		}

		arr = new int[N];
		sel = new int[M];
		queue = new LinkedList<>();
		sumqueue = new LinkedList<>();
		visited = new boolean[N];
		res = new int[2];
		arr2 = new int[M];

		for (int i = 0; i < N; i++) {
			arr[i] = i + 1;
		}

		combination(0, 0);

		int[] one_score = new int[cnt / 2];
		int[] two_score = new int[cnt / 2];
		int one_index = 0;
		int two_index = cnt / 2 - 1;

		int qsize = queue.size();

		for (int i = 0; i < (cnt); i++) {
			// 1팀
			if (i < (cnt / 2)) {
				for (int j = 0; j < M; j++) {
					arr2[j] = queue.poll();
				}

				perm(0);
				sumqueue.add(tmp);
				tmp = 0;

				if (one_index < cnt / 2) {
					one_score[one_index] = sumqueue.poll();
					one_index++;
				}

				for (int j = 0; j < M; j++) {
					arr2[j] = 0;
					visited[j] = false;
				}

				for (int j = 0; j < 2; j++) {
					res[j] = 0;
				}
			} else {
				// 2팀
				for (int j = 0; j < M; j++) {
					arr2[j] = queue.poll();
				}

				perm(0);
				sumqueue.add(tmp);
				tmp = 0;

				if (two_index >= 0) {
					two_score[two_index] = sumqueue.poll();
					two_index--;
				}
				for (int j = 0; j < M; j++)
					arr2[j] = 0;
				for (int j = 0; j < 2; j++) {
					res[j] = 0;
				}
			}

		}
		int min = Integer.MAX_VALUE;
		for (int i = 0; i < cnt / 2; i++) {
			min = Math.min(min, Math.abs(one_score[i] - two_score[i]));
		}
		bw.write(min + "");
		bw.close();
	}

	// 팀 나누기
	private static void combination(int idx, int s_idx) {

		if (s_idx == sel.length) {
			for (int i = 0; i < sel.length; i++) {
				queue.add(sel[i]);
			}
			cnt++;
			return;
		}

		for (int i = idx; i < arr.length; i++) {
			sel[s_idx] = arr[i];
			combination(i + 1, s_idx + 1);
		}
	}

  //팀 안에서 두 개씩 
	private static void perm(int idx) {
		if (idx == res.length) {
			tmp += map[res[0]][res[1]];
			return;
		}

		for (int i = 0; i < arr2.length; i++) {
			if (!visited[i]) {
				visited[i] = true;
				res[idx] = arr2[i];
				perm(idx + 1);
				visited[i] = false;
			}
		}
	}
}

```

### 문제점
처음에 팀을 나누기 위해 조합을 이용했다. 각 스타트와 링크팀 안에서도 2명씩의 합을 구해야 했기 때문에
또 순열을 이용하였다. 쉽게 짤 수 있는 방법을 찾아봐야겠다.
