## 10026.  적록색약

---

### Code

```java
import java.util.*;

class Pair {
	int x, y;

	Pair(int x, int y) {
		this.x = x;
		this.y = y;
	}
}

public class 적록색약 {
	static int N;
	static char[][] map, map2;
	static boolean[][] visited;
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };
	static Queue<Pair> queue;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		N = sc.nextInt();

		map = new char[N][N];
		map2 = new char[N][N];

		for (int i = 0; i < N; i++) {
			String str = sc.next();
			for (int j = 0; j < N; j++) {
				map[i][j] = str.charAt(j);
				map2[i][j] = str.charAt(j);
				if (map2[i][j] == 'G')
					map2[i][j] = 'R';
			}
		}
		int cnt = 0;

		visited = new boolean[N][N];

		// 적록색약은 빨강 초록 같은걸로 취급
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (!visited[i][j]) {
					bfs(i, j);
					cnt++;
				}
			}
		}

		visited = new boolean[N][N];

		int cnt2 = 0;

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (!visited[i][j]) {
					bfs2(i, j);
					cnt2++;
				}
			}
		}
		
		System.out.println(cnt+" "+cnt2);
	}

	private static void bfs(int r, int c) {
		queue = new LinkedList<>();
		queue.add(new Pair(r, c));

		while (!queue.isEmpty()) {

			Pair pair = queue.poll();

			for (int d = 0; d < 4; d++) {
				int nr = pair.x + dr[d];
				int nc = pair.y + dc[d];

				if (nr >= 0 && nc >= 0 && nr < N && nc < N) {
					if (!visited[nr][nc] && map[r][c] == map[nr][nc]) {
						visited[nr][nc] = true;
						queue.add(new Pair(nr, nc));
					}
				}
			}
		}
	}

	private static void bfs2(int r, int c) {
		queue = new LinkedList<>();
		queue.add(new Pair(r, c));

		while (!queue.isEmpty()) {

			Pair pair = queue.poll();

			for (int d = 0; d < 4; d++) {
				int nr = pair.x + dr[d];
				int nc = pair.y + dc[d];

				if (nr >= 0 && nc >= 0 && nr < N && nc < N) {
					if (!visited[nr][nc] && map2[r][c] == map2[nr][nc]) {
						visited[nr][nc] = true;
						queue.add(new Pair(nr, nc));
					}
				}
			}
		}
	}
}
```
