## 1697.  숨바꼭질

---

### Code

```java
import java.util.*;

public class Main {
	static int[] dr = { -1, 1, 2 };

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int N = sc.nextInt();
		int K = sc.nextInt();
		int[] arr = new int[100001];

		arr[N] = 1;
		arr[K] = 0;

		Queue<Integer> queue = new LinkedList<>();
		queue.add(N);

		int qsize = 0;
		int level = 0;
		int nr = 0;
		while (!queue.isEmpty()) {
			qsize = queue.size();
			for (int i = 0; i < qsize; i++) {
				int tmp = queue.poll();
				if (tmp == K) {
					System.out.println(level);
					return;
				}
				for (int d = 0; d < 3; d++) {
					if (d == 0 || d == 1)
						nr = tmp + dr[d];
					else if (d == 2) {
						nr = tmp * dr[d];
					}
					if (nr >= 0 && nr <= 100000) {
						if (arr[nr] == 0) {
							arr[nr] = 1;
							queue.add(nr);
						}

					}
				}
			}
			level++;
		}
	}
}
```
