## 1012.  유기농 배추

---

### Code

```java
import java.util.*;

public class Main {
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };
	static int[][] map;
	static int N, M;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int T = sc.nextInt();

		for (int tc = 1; tc <= T; tc++) {
			M = sc.nextInt(); // 가로 길이
			N = sc.nextInt(); // 세로 길이
			int K = sc.nextInt(); // 개수

			map = new int[N][M];

			for (int i = 0; i < K; i++) {
				int a = sc.nextInt();
				int b = sc.nextInt();
				map[b][a] = 1;
			}

			int cnt = 0;
			for (int i = 0; i < N; i++) {
				for (int j = 0; j < M; j++) {
					if (map[i][j] == 1) {
						dfs(i, j);
						cnt++;
					}

				}
			}
			System.out.println(cnt);

		}

	}

	static void dfs(int i, int j) {
		map[i][j] = 0;

		for (int d = 0; d < 4; d++) {
			int nr = i + dr[d];
			int nc = j + dc[d];
			if (nr >= 0 && nc >= 0 && nr < N && nc < M) {
				if (map[nr][nc] == 1) {
					dfs(nr, nc);
				}
			}
		}
	}
}

```

### BFS로 풀기

```java
import java.util.*;

class Node {
	int r, c;

	Node(int r, int c) {
		this.r = r;
		this.c = c;
	}
}

public class Main {
	static int N, M;
	static int[][] map;
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int T = sc.nextInt();

		for (int tc = 1; tc <= T; tc++) {
			M = sc.nextInt();
			N = sc.nextInt();
			int K = sc.nextInt();
			map = new int[N][M];

			for (int i = 0; i < K; i++) {
				int a = sc.nextInt();
				int b = sc.nextInt();
				map[b][a] = 1;
			}

			Queue<Node> queue = new LinkedList<Node>();
			int cnt = 0;
			for (int i = 0; i < N; i++) {
				for (int j = 0; j < M; j++) {
					if (map[i][j] == 1) {
						queue.add(new Node(i, j));
						cnt++;

						while (!queue.isEmpty()) {
							Node node = queue.poll();
							for (int d = 0; d < 4; d++) {
								int nr = node.r + dr[d];
								int nc = node.c + dc[d];
								if (nr >= 0 && nc >= 0 && nr < N && nc < M) {
									if (map[nr][nc] == 1) {
										map[nr][nc] = 0;
										queue.add(new Node(nr, nc));
									}

								}
							}

						}

					}

				}
			}
			System.out.println(cnt);
		}
	}
}
```
