## 1926.  그림

---

### Code

```java
package 백준_Solution;

import java.util.*;

class Pos {
	int r, c;

	Pos(int r, int c) {
		this.r = r;
		this.c = c;
	}
}

public class 그림 {
	static int N, M;
	static Queue<Pos> queue;
	static int[][] map;
	static boolean[][] visited;
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };
	static int cnt = 0;
	static int pic_num = 0;
	static PriorityQueue<Integer> pq;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		N = sc.nextInt();
		M = sc.nextInt();

		map = new int[N][M];
		queue = new LinkedList<>();
		pq = new PriorityQueue<>();
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				map[i][j] = sc.nextInt();
			}
		}

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				if (map[i][j] == 1) {
					bfs(i, j);
					pic_num++;
				}
			}
		}
		System.out.println(pic_num);
		int tmp = 0;
		while (!pq.isEmpty())
			tmp = pq.poll();
		if (pic_num == 0)
			System.out.println(0);
		else
			System.out.println(tmp);
	}

	private static void bfs(int i, int j) {
		map[i][j] = 0;
		queue.add(new Pos(i, j));
		while (!queue.isEmpty()) {
			Pos node = queue.poll();

			for (int d = 0; d < 4; d++) {
				int nr = node.r + dr[d];
				int nc = node.c + dc[d];

				if (nr >= 0 && nc >= 0 && nr < N && nc < M) {
					if (map[nr][nc] != 0) {
						map[nr][nc] = 0;
						queue.add(new Pos(nr, nc));
					}
				}
			}
			cnt++;
		}
		pq.add(cnt);
		cnt = 0;
	}
}
```
