## 15686.  치킨 배달

---

### Code

```java
import java.util.*;

class Pair {
	int x, y;

	Pair(int x, int y) {
		this.x = x;
		this.y = y;
	}
}

public class 치킨배달 {
	static int[][] map;
	static ArrayList<Pair> chickenlist, result;
	static int N, M, cnt;
	static Queue<Pair> queue;
	static int[][] visited;
	static int ans = Integer.MAX_VALUE;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		N = sc.nextInt(); // 도시 크기 NxN
		M = sc.nextInt(); // 치킨집 중 M개 선택

		map = new int[N][N];
		chickenlist = new ArrayList<>();

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				map[i][j] = sc.nextInt();
			}
		}

		// 0은 빈칸, 1은 집, 2는 치킨 집
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (map[i][j] == 2) {
					chickenlist.add(new Pair(i, j));
				}
			}
		}
		result = new ArrayList<Pair>();
		int final_ans = Integer.MAX_VALUE;
		for (int i = 1; i <= M; i++) {
			cnt = i;
			combination(0, 0);
			final_ans = Math.min(final_ans, ans);
		}
		System.out.println(final_ans);
	}

	static void combination(int depth, int idx) {
		if (depth == cnt) {
			ans = Math.min(ans, bfs());
			return;

		}
		for (int i = idx; i < chickenlist.size(); i++) {
			result.add(chickenlist.get(i));
			combination(depth + 1, i + 1);
			result.remove(result.size() - 1);
		}
	}

	static int bfs() {
		visited = new int[N][N];
		queue = new LinkedList<>();

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				visited[i][j] = Integer.MAX_VALUE;
			}
		}

		for (Pair p : result) {
			queue.add(p);
			visited[p.x][p.y] = 0;
		}

		int qsize = 0;
		int level = 0;
		int dist = 0;

		while (!queue.isEmpty()) {
			qsize = queue.size();
			for (int i = 0; i < qsize; i++) {
				Pair node = queue.poll();

				for (int j = 0; j < N; j++) {
					for (int k = 0; k < N; k++) {
						if (map[j][k] == 1) {
							int tmp = Math.abs(node.x - j) + Math.abs(node.y - k);
							visited[j][k] = Math.min(visited[j][k], tmp);
						} else {
							visited[j][k] = 0;
						}
					}
				}
			}
		}

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				dist += visited[i][j];
			}
		}
		return dist;
	}
}
```
