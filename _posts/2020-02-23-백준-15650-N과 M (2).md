## 15650.  N과 M (2)
---

### Code

```java
import java.util.*;

public class N과M2 {
	static int[] arr;
	static int[] sel;
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int N = sc.nextInt();
		int M = sc.nextInt();
		
		arr = new int[N];
		sel = new int[M];
		
		for(int i=0; i<N; i++) {
			arr[i] = i+1;
		}
		
		combination(0,0);
	}

	private static void combination(int idx, int s_idx) {
		if(s_idx==sel.length) {
			for(int i=0; i<sel.length; i++) {
				System.out.print(sel[i]+" ");
			}
			System.out.println();
			return;
		}
		
		for(int i=idx; i<arr.length; i++) {
			sel[s_idx] = arr[i];
			combination(i+1, s_idx+1);
		}
	}
}

```
