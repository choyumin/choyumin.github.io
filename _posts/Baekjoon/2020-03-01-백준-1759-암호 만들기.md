## 1759.  암호 만들기

---

### Code

```java
import java.util.*;

class Alpha implements Comparable<Alpha> {
	char alphabet;
	int idx;

	Alpha(char alphabet, int idx) {
		this.alphabet = alphabet;
		this.idx = idx;
	}

	@Override
	public int compareTo(Alpha o) {
		if (idx < o.idx)
			return -1;
		else if (idx > o.idx)
			return 1;
		return 0;
	}

}

public class 암호만들기 {
	static ArrayList<Alpha> array, result;
	static int L, C;
	static boolean[] visited;
	static StringBuilder sb;
	static int vowel = 0;
	static int consonant = 1;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		L = sc.nextInt(); // sel 개수
		C = sc.nextInt(); // arr 개수

		array = new ArrayList<>();
		result = new ArrayList<>();
		sb = new StringBuilder();

		for (int i = 0; i < C; i++) {
			String str = sc.next();
			array.add(new Alpha(str.charAt(0), (int) str.charAt(0)));
		}

		Collections.sort(array);

		comb(0, 0);
	}

	private static void comb(int depth, int idx) {

		if (depth == L) {
			for (Alpha a : result) {
				sb.append(a.alphabet);
				if (a.alphabet == 'a' || a.alphabet == 'e' || a.alphabet == 'i' || a.alphabet == 'o'
						|| a.alphabet == 'u')
					vowel++;
				else
					consonant++;

			}
			if (vowel >= 1 && consonant >= 2)
				System.out.println(sb.toString());
			vowel = 0;
			consonant = 0;
			sb.delete(0, result.size());
			return;
		}

		for (int i = idx; i < array.size(); i++) {
			result.add(array.get(i));
			comb(depth + 1, i + 1);
			result.remove(result.size() - 1);
		}
	}
}
```
