## 2667.  단지번호붙이기

---

### Code

```java
import java.util.*;

public class Main {
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };
	static int[][] arr;
	static int N;
	static int danji_cnt;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		N = sc.nextInt();
		arr = new int[N][N];

		for (int i = 0; i < N; i++) {
			String str = sc.next();
			for (int j = 0; j < N; j++) {
				arr[i][j] = str.charAt(j) - '0';
			}
		}
		int cnt = 0; // 단지 수
		PriorityQueue<Integer> queue = new PriorityQueue<>();

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (arr[i][j] == 1) {
					dfs(i, j);
					cnt++;
					queue.add(danji_cnt + 1);
					danji_cnt = 0;
				}

			}
		}
		System.out.println(cnt);
		for (int i = 0; i < cnt; i++) {
			System.out.println(queue.poll());
		}

	}

	static void dfs(int i, int j) {
		arr[i][j] = 0;

		for (int d = 0; d < 4; d++) {
			int nr = i + dr[d];
			int nc = j + dc[d];
			if (nr >= 0 && nc >= 0 && nr < N && nc < N) {
				if (arr[nr][nc] == 1) {
					dfs(nr, nc);
					danji_cnt++;
				}

			}
		}
	}
}
```

### BFS 이용

```java
import java.util.*;

class Node {
	int r, c;

	Node(int r, int c) {
		this.r = r;
		this.c = c;
	}
}

public class Main {
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int N = sc.nextInt();
		int[][] map = new int[N][N];

		for (int i = 0; i < N; i++) {
			String str = sc.next();
			for (int j = 0; j < N; j++) {
				map[i][j] = str.charAt(j) - '0';
			}
		}

		Queue<Node> queue = new LinkedList<>();
		PriorityQueue<Integer> prqueue = new PriorityQueue<>();
		int cnt = 0;

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (map[i][j] == 1) {
					map[i][j] = 0;
					cnt++;
					int danji_cnt = 0;
					queue.add(new Node(i, j));

					while (!queue.isEmpty()) {
						Node node = queue.poll();
						danji_cnt++;
						for (int d = 0; d < 4; d++) {
							int nr = node.r + dr[d];
							int nc = node.c + dc[d];

							if (nr >= 0 && nc >= 0 && nr < N && nc < N) {
								if (map[nr][nc] == 1) {
									map[nr][nc] = 0;
									queue.add(new Node(nr, nc));
								}
							}
						}

					}
					prqueue.add(danji_cnt);
				}
			}
		}
		System.out.println(cnt);
		for (int i = 0; i < cnt; i++) {
			System.out.println(prqueue.poll());
		}

	}
}
```
